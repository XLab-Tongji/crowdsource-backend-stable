post /api/account/register

创建用户

  • Request
  • Response

Body

Type: application/json

Schema:

{
    "type":"object",
    "required": true,
    "properties":{
        "username":{
                "type":"string",
                "required": true
        },

        "name":{
                "type":"string",
                "required":true
        },
        "password":{
                "type":"string",
                "required":true
        },
        "mobile":{
                "type":"string",
                "required":true
        },
        "create_time":{
                "type":"timestamp",
                "required":true
        },
        "update_time":{
                "type":"timestamp",
                "required":true
        },
        "email":{
                "type":"string",
                "required":true
        },
        "ext_params":{
                "type":"json",
                "required":false
        }

    }
}
